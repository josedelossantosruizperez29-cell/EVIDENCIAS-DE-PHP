abstract class Vehiculo implements Conducible
{
    private static int $contador = 0;

    public function __construct(
        protected string $marca,
        protected string $modelo,
        protected int    $anio,
        protected float  $precio
    ) {
        self::$contador++;
    }

    // Métodos abstractos que cada hija debe implementar
    abstract public function getTipo():           string;
    abstract public function getIcono():          string;
    abstract public function getColor():          string;
    abstract public function consumoPor100km():   float;

    // Implementación del contrato: usa el método abstracto
    public function calcularCombustible(float $km): float
    {
        return round(($km / 100) * $this->consumoPor100km(), 2);
    }

    public function getMarca():  string { return $this->marca;  }
    public function getModelo(): string { return $this->modelo; }
    public function getAnio():   int    { return $this->anio;   }
    public function getPrecio(): float  { return $this->precio; }

    public function getPrecioFormateado(): string
    {
        return '$' . number_format($this->precio, 0, ',', '.');
    }

    public static function getContador(): int
    {
        return self::$contador;
    }

    public function __toString(): string
    {
        return sprintf('[%s] %s %s %d',
            $this->getTipo(), $this->marca, $this->modelo, $this->anio);
    }
}

class Auto extends Vehiculo
{
    public function __construct(
        string $marca, string $modelo, int $anio, float $precio,
        private int $puertas = 4
    ) {
        parent::__construct($marca, $modelo, $anio, $precio);
    }

    public function getTipo():         string { return 'Auto'; }
    public function getIcono():        string { return 'fas fa-car'; }
    public function getColor():        string { return '#3b82f6'; }
    public function consumoPor100km(): float  { return 8.5; }

    public function arrancar(): string
    {
        return "El Auto {$this->marca} {$this->modelo} arranca con la llave.";
    }

    public function frenar(): string
    {
        return "El Auto {$this->marca} frena con discos de freno.";
    }

    public function getPuertas(): int { return $this->puertas; }
}
⚠ No olvides: siempre llama parent::__construct() o el contador estático de Vehiculo no se incrementará.
